# 图片压缩

  • 主要使用了canvas的drawImage、toDataURL、toBolb 的 API
  • 压缩方向主要从图片的宽高和质量进行压缩

WARNING

如果压缩格式为png，则图片的质量不会进行压缩

原图: 文件大小：0B, base64格式大小：0B

base64压缩后的图片: 大小：0B, 压缩比：NaN

Blob压缩后的图片 大小：0B, 压缩比：NaN

  <script>
    const uploadEle = document.getElementById('upload')
    const ACCEPT = ['image/png', 'image/jpg', 'image/jpeg']
    const MAX_SIZE = 4 * 1024 * 1024
    const MAXSIZE_STR = '4MB'
    let fileSize = 0
    uploadEle.addEventListener('change', function (e) {
      const [file] = e.target.files
      if (!file) return
      const { type: fileType } = file
      fileSize = file.size
      // 检测文件类型
      if (ACCEPT.indexOf(fileType) < 0) {
        alert('不支持上传' + fileType + '文件类型')
        return
      }
      // 检测文件大小
      if (MAX_SIZE < fileSize) {
        alert(`文件超出${MAXSIZE_STR}！`)
        return
      }
      //图片压缩
      convertImageToBase64(file, (base64Image) =>
        compress(base64Image, uploadToServer)
      )
    })
    // 将图片转成base64格式
    function convertImageToBase64(file, callback) {
      let reader = new FileReader()
      reader.addEventListener('load', function (e) {
        const base64Image = e.target.result
        callback && callback(base64Image)
        // 将reader资源进行回收
        reader = null
      })
      reader.readAsDataURL(file)
    }
    function compress(base64Image, callback) {
      const srcImgEle = document.createElement('img')
      srcImgEle.src = base64Image
      srcImgEle.addEventListener('load', function (e) {
        let maxW = 1024 //实际最大宽度
        let maxH = 1024 //实际最大高度
        let radio = 1 //比例
        let needCompress = false //是否需要进行压缩
        // 展示使用
        document.body.append(srcImgEle)
        if (maxW < srcImgEle.naturalWidth) {
          needCompress = true
          radio = srcImgEle.naturalWidth / maxW
          maxH = srcImgEle.naturalHeight / radio
        }
        // 如果上面已经进行宽度比设置后，这里就不用再执行了
        if (maxH < srcImgEle.naturalHeight && !needCompress) {
          needCompress = true
          radio = srcImgEle.naturalHeight / maxH
          maxW = srcImgEle.naturalWidth / radio
        }
        // 如果不需要进行压缩则最大宽高等于图片的实际宽高
        if (!needCompress) {
          maxW = srcImgEle.naturalWidth
          maxH = srcImgEle.naturalHeight
        }
        // 创建canvas进行图片压缩
        let canvas = document.createElement('canvas')
        let ctx = canvas.getContext('2d')
        canvas.width = maxW
        canvas.height = maxH
        ctx.drawImage(srcImgEle, 0, 0, maxW, maxH)
        /**
         * 将图片转成base64 如果直接保存成字符串的话，体积相对会比较大，但保存成文件则跟blob的大小一样
         * 第一个参数 转换后的图片类型
         * 第二个参数 转换后的图片质量 对要输出为png的图片类型无效
         */
        const compressImage = canvas.toDataURL('image/png', 0.9)
        callback && callback(compressImage)
        console.log(
          '这里base64的压缩比：' + base64Image.length / compressImage.length
        )
        //这段代码仅仅是做展示使用
        const targetImgEle = new Image()
        targetImgEle.src = compressImage
        document.body.append(targetImgEle)
        /**
         * 将图片转成blob文件 与base64的上传方法应该不一样,这两种方式可以选择其中一种都行
         * 第一个参数是回调函数
         * 第二个参数 转换后的图片类型
         * 第三个参数 转换后的图片质量 对要输出为png的图片类型无效
         */
        const compressImage2 = canvas.toBlob(
          function (blob) {
            const targetImgEle = new Image()
            // 展示使用
            // targetImgEle.src = URL.createObjectURL(blob)
            // document.body.append(targetImgEle)
            console.log(blob.size)
            console.log(fileSize)
            console.log('这里是文件的压缩比：' + fileSize / blob.size)
          },
          'image/png',
          0.9
        )
        canvas = null
        ctx = null
      })
    }
    // 上传图片的方法，根据实际情况进行编写
    function uploadToServer(compressImage) {
      console.log('upload to server...', compressImage)
    }
  </script>

<template>
  <el-card>
    <input type="file" @change="onChange" />
    <div>
      <h3>
        原图: 文件大小：{{ fileSize | fileSizeFormat }}, base64格式大小：{{
          base64Size | fileSizeFormat
        }}
      </h3>
      <el-image :src="srcList[0]" :preview-src-list="srcList"> </el-image>
    </div>
    <div>
      <h3>
        base64压缩后的图片: 大小：{{ compressBase64Size | fileSizeFormat }},
        压缩比：{{ base64Size / compressBase64Size }}
      </h3>
      <el-image :src="srcList[1]" :preview-src-list="srcList"> </el-image>
      <h3>
        Blob压缩后的图片 大小：{{ compressBlobSize | fileSizeFormat }},
        压缩比：{{ fileSize / compressBlobSize }}
      </h3>
      <el-image :src="srcList[2]" :preview-src-list="srcList"> </el-image>
    </div>
  </el-card>
</template>

<script>
export default {
  name: 'ImageCompress',
  data() {
    this.MAX_SIZE = 1024 * 1024 * 4
    this.ACCEPT = ['image/png', 'image/jpg', 'image/jpeg']
    this.MAXSIZE_STR = '4MB'

    return {
      fileSize: 0,
      base64Size: 0,
      compressBase64Size: 0,
      compressBlobSize: 0,
      srcList: [],
    }
  },
  methods: {
    onChange(e) {
      const [file] = e.target.files
      // 图片类型
      const fileType = file.type
      // 图片大小
      this.fileSize = file.size
      if (this.ACCEPT.indexOf(fileType) < 0) {
        this.$message.error(`不支持上传${fileType}类型文件`)
      }
      if (this.fileSize > this.MAX_SIZE) {
        this.$message.error(`文件超出${this.MAXSIZE_STR}!`)
      }
      // 图片压缩
      this.convertImageToBase64(file, (base64Image) => {
        this.compress(base64Image, this.uploadToServer)
      })
    },
    convertImageToBase64(file, callback) {
      let reader = new FileReader()
      let that = this
      reader.addEventListener('load', function (e) {
        const base64Image = e.target.result
        that.base64Size = base64Image.length
        that.$set(that.srcList, 0, base64Image)
        reader = null
        callback && callback(base64Image)
      })
      reader.readAsDataURL(file)
    },
    compress(base64Image, callback) {
      let that = this
      let srcImageEle = new Image()
      srcImageEle.src = base64Image
      srcImageEle.addEventListener('load', function (e) {
        let maxW = 1024
        let maxH = 1024
        let radio = 1
        let needCompress = false
        const nW = srcImageEle.naturalWidth
        const nH = srcImageEle.naturalHeight
        if (maxW < nW) {
          needCompress = true
          radio = nW / maxW
          maxH = nH / radio
        }
        if (maxH < nH && !needCompress) {
          needCompress = true
          radio = nH / maxH
          maxW = nW / radio
        }
        if (!needCompress) {
          maxW = nW
          maxH = nH
        }
        let canvas = document.createElement('canvas')
        let ctx = canvas.getContext('2d')
        canvas.width = maxW
        canvas.height = maxH
        ctx.drawImage(srcImageEle, 0, 0, maxW, maxH)
        /**
         * 将图片转成base64 如果直接保存成字符串的话，体积相对会比较大，但保存成文件则跟blob的大小一样
         * 第一个参数 转换后的图片类型
         * 第二个参数 转换后的图片质量 对要输出为png的图片类型无效
         */
        const compressImage = canvas.toDataURL('image/png', 0.9)
        callback && callback(compressImage)
        that.$set(that.srcList, 1, compressImage)
        that.compressBase64Size = compressImage.length
        /**
         * 将图片转成blob文件 与base64的上传方法应该不一样,这两种方式可以选择其中一种都行
         * 第一个参数是回调函数
         * 第二个参数 转换后的图片类型
         * 第三个参数 转换后的图片质量 对要输出为png的图片类型无效
         */
        canvas.toBlob(
          function (blob) {
            that.$set(that.srcList, 2, URL.createObjectURL(blob))
            that.compressBlobSize = blob.size
          },
          'image/png',
          0.9
        )
        srcImageEle = null
        canvas = null
        ctx = null
      })
    },
    uploadToServer(compressImage) {
      console.log(`upload to server...${compressImage}`)
    },
  },
}
</script>
